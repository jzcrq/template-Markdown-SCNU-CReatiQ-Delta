## DEBUG 头

```cpp
#include <bits/stdc++.h>
using namespace std;
using i64=long long;
using i128=__int128;

namespace DBG
{
	template <class T>
	void _dbg(const char *f,T t) { cerr<<f<<'='<<t<<'\n'; }

	template <class A,class... B>
	void _dbg(const char *f,A a,B... b)
	{
		while (*f!=',') cerr<<*f++;
		cerr<<'='<<a<<",";
		_dbg(f+1,b...);
	}

	template <class T>
	ostream& operator << (ostream& os,const vector<T> &v)
	{
		os<<"[ ";
		for (const auto &x:v) os<<x<<", ";
		os<<"]";
		return os;
	}

	#define dbg(...) _dbg(#__VA_ARGS__, __VA_ARGS__)
}

using namespace DBG;
```

## __int128 输入输出流

```cpp
istream &operator >> (istream &is,i128 &x)
{
	string s;
	is>>s;
	bool neg=0;
	x=0;
	for (char c:s)
	{
		if (c=='-') neg=1;
		else x=x*10+(c-'0');
	}
	if (neg) x=-x;
	return is;
}

ostream &operator << (ostream &os,i128 x)
{
	if (x==0) os<<0;
	else
	{
		string s,t;
		if (x<0)
		{
			x=-x;
			t='-';
		}
		while (x)
		{
			s.push_back('0'+x%10);
			x/=10;
		}
		reverse(s.begin(),s.end());
		os<<t<<s;
	}
	return os;
}
```

## 常用数学函数

```cpp
i64 ceilDiv(i64 n,i64 m)
{
	if (n>=0) return (n+m-1)/m;
	else return n/m;
}

i64 floorDiv(i64 n,i64 m)
{
	if (n>=0) return n/m;
	else return (n-m+1)/m;
}

i128 gcd(i128 a,i128 b)
{
	return b?gcd(b,a%b):a;
}
```
